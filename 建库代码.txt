create table 个人报名表
(序号 int(11) not null primary key,                                 
姓名 varchar(15) not null,
性别 enum('boy','girl'),                                            
学号 int(15) not null,
院系 varchar(40),                                                   
班级专业 varchar(10),
手机号码 int(12) not null);


create table 管理者信息表
(序号 int(11) not null primary key auto_increment,                      
姓名 varchar(15) not null,
性别 enum('boy','girl'),                                             
学号 int(15) not null,                                           
院系 varchar(40),
班级专业 varchar(10),
手机号码 int(12) not null,
组织名 varchar(10) not null,
职位 varchar(10)                                                 
);


create table 活动信息表
(活动名 varchar(30) not null,
活动内容 longtext not null,
活动类型 varchar(50),                                              
发布时间 datetime not null,                                        
活动状态 enum('not_push','In','out_of_date'),
活动组织归属 varchar(10),
活动截止时间 datetime,                                             
接受报名时间 datetime);                                                                                          


create table 通知信息表
(信息名 varchar(25),
通知内容 longtext not null,
通知时间 datetime,                                                      
发布组织 varchar(25) not null);


create table 团体报名表
(序号 int(11) not null primary key,                                          
队长姓名 varchar(15),
队长联系方式 int(12) not null,
队伍名称 varchar(15) not null,
队伍人数 int(5),
队长学号 int(15),                                                         
队员姓名 text);                                                          

create table 用户信息表
(序号 int(11) not null primary key auto_increment,                         
姓名 varchar(15) not null,
性别 enum('boy','girl'),
学号 int(15) not null,
院系 varchar(40),
班级专业 varchar(10),
手机号码 int(12) not null,
微信号 varchar(25));                                                    

create table 账号信息表
(序号 int(11) primary key not null,
账号名 varchar(25) not null,
密码 varchar(16) not null);

create table 组织信息表
(组织名 varchar(15) not null,
组织性质 varchar(20),                                                        
指导老师姓名 varchar(25),                                                    
负责人姓名 varchar(5)not null);
